```typescript
// src/components/common/AnimatedSection.tsx
import { motion } from "motion/react";
import type { ReactNode } from "react";
import { fadeUp } from "@/lib/animations";
import { cn } from "@/lib/cn";

interface AnimatedSectionProps {
  children: ReactNode;
  delay?: number;
  className?: string;
}

export default function AnimatedSection({
  children,
  delay = 0,
  className,
}: AnimatedSectionProps) {
  return (
    <motion.div
      className={cn(className)}
      variants={fadeUp}
      initial="hidden"
      whileInView="visible"
      viewport={{ once: true, margin: "-80px" }}
      transition={{ duration: 0.5, ease: "easeOut", delay }}
    >
      {children}
    </motion.div>
  );
}
```

Key decisions:
- Import from `"motion/react"` (not `"framer-motion"`) — forward-compatible path per STATE.md
- `viewport.once: true` prevents replay on scroll-back (per STATE.md architecture decision)
- `viewport.margin: "-80px"` triggers animation slightly before element enters viewport for better UX
- Uses `fadeUp` variant from animations.ts — single source of truth for animation values
- `cn()` wrapper handles className merging correctly with Tailwind

```typescript
// src/hooks/useScrollLock.ts
import { useEffect } from "react";

// Module-level reference count — survives across component instances
const lockCount = { value: 0 };

export function useScrollLock(active: boolean) {
  useEffect(() => {
    if (active) {
      lockCount.value++;
      document.body.style.overflow = "hidden";
    }
    return () => {
      if (active) {
        lockCount.value--;
        if (lockCount.value === 0) {
          document.body.style.overflow = "";
        }
      }
    };
  }, [active]);
}
```

Key decisions:
- Reference-counted: multiple consumers can lock scroll simultaneously without race condition
- Module-level `lockCount` object (not primitive) — survives HMR and component remounts
- Only restores overflow when ALL consumers have unlocked (count reaches 0)
- Replaces v1's bug-prone `document.body.style.overflow = "hidden"` with setTimeout(900ms) pattern
- This hook will be consumed in Phase 2 when ProjectModal and HomePage are migrated

Do NOT modify any existing components in this task. Existing components still use their old scroll lock pattern. Migration happens in Phase 2.
